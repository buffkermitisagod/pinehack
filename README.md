# pinehack
a simple python hacking script made for the pine phone running linux (BETA NOT ALL TOOLS ARE INSTALLED AND TESTED)

# install
for pacman
    pacman -S git
    git clone https://github.com/buffkermitisagod/pinehack
    cd pinehack
    python install.py
for apt
    apt-get install git
    git clone https://github.com/buffkermitisagod/pinehack
    cd pinehack
    python install.py
